public class Article{

private String id_number;
private String date_accquired;
private int amount_of_copies;
private String description;
private String title;

public()Article{

title="";
description="";
amount_of_copies="";
date_accquired="";
id_number="";
}

public void setIdNumber(String temp){
id_number=temp;
}

public void setDateAccquired(String temp){
date_accquired=temp;
}

public void setAmountOfCopies(int temp){
amount_of_copies=temp;
}

public void setDescription(String temp){
description=temp;
}

public void setTitle(String temp){
title=temp;
}

public String getIdNumber(){
return id_number;
}

public String getDateAccquired(){
return date_accquired;
}

public int getAmountOfCopies(){
return amount_of_copies;
}

public String getDescription(){
return description;
}

public String getTitle(){
return title;
}

}
